class Solution:
    def invertTree(self, root: TreeNode) -> TreeNode:
        if not root:

        root.left,root.right = root.right,root.left  
        self.invertTree(root.left)
        self.invertTree(root.right)
        return root

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def invertTree(self, root: TreeNode) -> TreeNode:
        if not root:
            return root
        queue = [root]
        while queue:
            node = queue.pop(0)
            node.left,node.right = node.right,node.left
            if node.left:
                queue.append(node.left)
            if node.right:
                queue.append(node.right)
        return root

#快速幂+（递归回溯）
class Solution:
    def myPow(self, x: float, n: int) -> float:
        def quikenum(N):
            if N == 0:
                return 1.0
            y = quikenum(N//2)  #y指向最终结果的中间位置
            return y*y if N%2 == 0 else y*y*x #如果n为偶数，最终结果是y*y,如果是奇数得再乘以一个x
        return quikenum(n) if n>0 else (1.0/quikenum(-n))  #如果n为负数，按照-n计算再求倒数
    
    
#分治思路：
#pow(x,n):
#	subproblem pow(x,n/2)
#if n%2 ==1:
#	result = subresult *subresult*x
#else:
#	result = subresult *subresult

    
#快速幂+迭代
class Solution:
    def myPow(self, x: float, n: int) -> float:

        def quikemul(N):
            ans= 1      #承接所有贡献
            x_att = x #初始贡献
            while N >0: 
                if N%2 ==1: 
                    ans *= x_att #如果 N是奇数，多贡献一个x
                x_att *= x_att   #每次迭代都要将贡献平方
                N//=2            #N折半
            return ans           
        return quikemul(n) if n>=0 else 1.0/quikemul(-n)
